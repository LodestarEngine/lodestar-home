98template<typename TScalar, int TStateDim, int TInputDim, int TOutputDim>
99void ls::systems::DiscreteSystem<ls::systems::StateSpace<TScalar, TStateDim, TInputDim, TOutputDim>>::advance()
100{
101 if (input != nullptr) {
102 *state = (*system->getA()) * (*state) + (*system->getB()) * (*input);
103 } else {
104 *state = (*system->getA()) * (*state);
105 }
106
107 time += system->getSamplingPeriod();
108}
